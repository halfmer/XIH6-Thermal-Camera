# 热成像数据传输协议 & 平台指南

> 上位机支持 **串口** 和 **WiFi (TCP)** 两种传输方式，二进制帧协议完全统一。

---

## 一、二进制帧协议（通用）

### 帧结构

```
┌────────┬────────┬──────┬──────────┬───────┬────────┬───────────┬──────────┬──────────┐
│ 0xAA   │ 0x55   │ type │ frame_id │ width │ height │ pixel_len │  pixels  │ checksum │
│ 1 B    │ 1 B    │ 1 B  │   2 B    │  2 B  │  2 B   │    4 B    │   N B    │   2 B    │
└────────┴────────┴──────┴──────────┴───────┴────────┴───────────┴──────────┴──────────┘
```

| 偏移 | 字节数 | 字段 | 类型 | 说明 |
|------|--------|------|------|------|
| 0 | 1 | sync0 | u8 | 固定 `0xAA` |
| 1 | 1 | sync1 | u8 | 固定 `0x55` |
| 2 | 1 | type | u8 | `0x01` = raw 16-bit thermal |
| 3 | 2 | frame_id | u16 BE | 帧序号，上位机用于统计丢失 |
| 5 | 2 | width | u16 BE | LEPTON 3.5 = `160` |
| 7 | 2 | height | u16 BE | LEPTON 3.5 = `120` |
| 9 | 4 | pixel_len | u32 BE | = width × height × 2 |
| 13 | N | pixels | u16 BE[] | 原始像素数组，每像素 16-bit 大端 |
| 13+N | 2 | checksum | u16 BE | sync0 到 pixels 末字节的累加和 |

### 帧大小

| 传感器 | 分辨率 | 像素字节 | 帧总字节 |
|--------|--------|----------|----------|
| LEPTON 3.5 | 160×120 | 38,400 B | **38,413 B** |
| 自定义 | W×H | W×H×2 | W×H×2 + 13 |

### 校验和

```c
uint16_t checksum = 0;
for (int i = 0; i < total_bytes_before_checksum; i++)
    checksum += buffer[i];
```

## 二、ESP32-S3 — WiFi 传输（推荐）

### 连接拓扑

```
┌──────────────┐         WiFi (TCP)        ┌──────────────────┐
│  ESP32-S3    │ ═══════════════════════   │  PC 上位机        │
│  + LEPTON    │   ESP32 → PC:8888         │  TCP Server       │
│              │   二进制帧流                │  监听 0.0.0.0:8888│
└──────────────┘                           └──────────────────┘
```

ESP32 作为 WiFi Station 连接到 PC 所在局域网，然后以 TCP Client 身份连接 PC 的 IP:端口。

### ESP32 发送代码（ESP-IDF）

```c
#include "esp_wifi.h"
#include "esp_event.h"
#include "lwip/sockets.h"

#define THERMAL_W        160
#define THERMAL_H        120
#define TCP_SERVER_IP    "192.168.1.100"   // PC 的 IP（看上位机显示的 IP）
#define TCP_SERVER_PORT  8888

static int g_sock = -1;
static uint16_t g_frame_id = 0;

// ── 连接 WiFi ──────────────────────────────────────────────
void wifi_connect(const char *ssid, const char *password) {
    // 用 ESP-IDF WiFi station 示例连接路由器
    // 确保 ESP32 和 PC 在同一子网
}

// ── 连接 TCP Server ────────────────────────────────────────
bool tcp_connect_to_host(void) {
    g_sock = socket(AF_INET, SOCK_STREAM, IPPROTO_TCP);
    if (g_sock < 0) return false;

    // 禁用 Nagle 算法，降低延迟
    int flag = 1;
    setsockopt(g_sock, IPPROTO_TCP, TCP_NODELAY, &flag, sizeof(flag));

    struct sockaddr_in server_addr = {
        .sin_family = AF_INET,
        .sin_port   = htons(TCP_SERVER_PORT),
    };
    inet_pton(AF_INET, TCP_SERVER_IP, &server_addr.sin_addr);

    if (connect(g_sock, (struct sockaddr *)&server_addr, sizeof(server_addr)) != 0) {
        close(g_sock);
        g_sock = -1;
        return false;
    }
    return true;
}

// ── 发送一帧热成像数据 ─────────────────────────────────────
void sendThermalFrame(uint8_t *frame_buffer) {
    if (g_sock < 0) return;

    uint32_t pixel_len = THERMAL_W * THERMAL_H * 2;
    uint16_t cksum = 0;
    uint8_t buf[16];

    // Sync
    buf[0] = 0xAA; buf[1] = 0x55;
    send(g_sock, buf, 2, 0);
    cksum += 0xAA + 0x55;

    // Header (11 bytes)
    uint16_t fid = g_frame_id++;
    buf[0]  = 0x01;
    buf[1]  = (fid >> 8) & 0xFF;
    buf[2]  = fid & 0xFF;
    buf[3]  = (THERMAL_W >> 8) & 0xFF;
    buf[4]  = THERMAL_W & 0xFF;
    buf[5]  = (THERMAL_H >> 8) & 0xFF;
    buf[6]  = THERMAL_H & 0xFF;
    buf[7]  = (pixel_len >> 24) & 0xFF;
    buf[8]  = (pixel_len >> 16) & 0xFF;
    buf[9]  = (pixel_len >> 8) & 0xFF;
    buf[10] = pixel_len & 0xFF;
    send(g_sock, buf, 11, 0);
    for (int i = 0; i < 11; i++) cksum += buf[i];

    // Pixels
    send(g_sock, frame_buffer, pixel_len, 0);
    for (uint32_t i = 0; i < pixel_len; i++) cksum += frame_buffer[i];

    // Checksum
    buf[0] = (cksum >> 8) & 0xFF;
    buf[1] = cksum & 0xFF;
    send(g_sock, buf, 2, 0);
}
```

### Arduino 版 WiFi 发送

```cpp
#include <WiFi.h>

#define THERMAL_W        160
#define THERMAL_H        120
#define TCP_SERVER_IP    "192.168.1.100"
#define TCP_SERVER_PORT  8888

static WiFiClient g_client;
static uint16_t g_frame_id = 0;

void setup() {
    WiFi.begin("YOUR_SSID", "YOUR_PASSWORD");
    while (WiFi.status() != WL_CONNECTED) delay(500);

    if (g_client.connect(TCP_SERVER_IP, TCP_SERVER_PORT)) {
        g_client.setNoDelay(true);  // 禁用 Nagle
    }
}

void sendThermalFrame(uint8_t *frame_buffer) {
    if (!g_client.connected()) {
        g_client.connect(TCP_SERVER_IP, TCP_SERVER_PORT);
        g_client.setNoDelay(true);
    }

    uint32_t pixel_len = THERMAL_W * THERMAL_H * 2;
    uint16_t cksum = 0;

    g_client.write(0xAA); cksum += 0xAA;
    g_client.write(0x55); cksum += 0x55;
    g_client.write(0x01); cksum += 0x01;

    uint16_t fid = g_frame_id++;
    g_client.write((fid >> 8) & 0xFF); cksum += (fid >> 8) & 0xFF;
    g_client.write(fid & 0xFF);        cksum += fid & 0xFF;

    g_client.write((THERMAL_W >> 8) & 0xFF); cksum += (THERMAL_W >> 8) & 0xFF;
    g_client.write(THERMAL_W & 0xFF);        cksum += THERMAL_W & 0xFF;
    g_client.write((THERMAL_H >> 8) & 0xFF); cksum += (THERMAL_H >> 8) & 0xFF;
    g_client.write(THERMAL_H & 0xFF);        cksum += THERMAL_H & 0xFF;

    g_client.write((pixel_len >> 24) & 0xFF); cksum += (pixel_len >> 24) & 0xFF;
    g_client.write((pixel_len >> 16) & 0xFF); cksum += (pixel_len >> 16) & 0xFF;
    g_client.write((pixel_len >> 8)  & 0xFF); cksum += (pixel_len >> 8)  & 0xFF;
    g_client.write(pixel_len & 0xFF);         cksum += pixel_len & 0xFF;

    g_client.write(frame_buffer, pixel_len);
    for (uint32_t i = 0; i < pixel_len; i++) cksum += frame_buffer[i];

    g_client.write((cksum >> 8) & 0xFF);
    g_client.write(cksum & 0xFF);
}
```

### ESP32 WiFi 注意事项

| ⚠️ 注意 | 说明 |
|----------|------|
| **TCP_NODELAY** | 必须禁用 Nagle 算法，否则 38KB 帧会被 TCP 延迟合并导致帧率下降 |
| **WiFi 信号强度** | 38KB/帧 @ 9fps ≈ 2.7 Mbps 持续吞吐。WiFi 信号差时会丢包，TCP 自动重传引入延迟 |
| **帧 buffer** | 38KB 放栈上会爆，用 `heap_caps_malloc` 或 `static` 全局分配 |
| **断线重连** | TCP 断开会自动触发重连，上位机恢复监听 |
| **PC 防火墙** | Windows 防火墙可能拦截 TCP 8888 端口，首次运行时需允许 |
| **IP 地址** | PC 的 IP 可能变化（DHCP），上位机会显示当前 IP，ESP32 端需硬编码或做 mDNS 发现 |

## 三、ESP32-S3 / STM32 — 串口传输（备选）

### 关键参数

| 项目 | 值 |
|------|-----|
| **波特率** | `921600`（推荐），最高 `3000000` |
| **数据位** | 8 |
| **停止位** | 1 |
| **校验位** | 无 |
| **流控** | 无 |

### 串口发送通用代码框架

```c
// 任意平台，只要能把字节发到串口即可
void uart_write(const uint8_t *data, uint32_t len);  // 平台相关的串口发送函数

void sendThermalFrame(uint8_t *frame_buffer) {
    uint32_t pixel_len = THERMAL_W * THERMAL_H * 2;
    uint16_t cksum = 0;
    uint8_t buf[16];

    buf[0] = 0xAA; buf[1] = 0x55;
    uart_write(buf, 2);  cksum += 0xAA + 0x55;

    uint16_t fid = g_frame_id++;
    buf[0]  = 0x01;
    buf[1]  = (fid >> 8) & 0xFF;
    buf[2]  = fid & 0xFF;
    buf[3]  = (THERMAL_W >> 8) & 0xFF;
    buf[4]  = THERMAL_W & 0xFF;
    buf[5]  = (THERMAL_H >> 8) & 0xFF;
    buf[6]  = THERMAL_H & 0xFF;
    buf[7]  = (pixel_len >> 24) & 0xFF;
    buf[8]  = (pixel_len >> 16) & 0xFF;
    buf[9]  = (pixel_len >> 8) & 0xFF;
    buf[10] = pixel_len & 0xFF;
    uart_write(buf, 11);
    for (int i = 0; i < 11; i++) cksum += buf[i];

    uart_write(frame_buffer, pixel_len);
    for (uint32_t i = 0; i < pixel_len; i++) cksum += frame_buffer[i];

    buf[0] = (cksum >> 8) & 0xFF;
    buf[1] = cksum & 0xFF;
    uart_write(buf, 2);
}
```

### 串口注意事项

| ⚠️ 注意 | 说明 |
|----------|------|
| **独占总线** | 热成像串口**只发帧**，`printf`/日志走另一个 UART |
| **VoSPI 字节序** | LEPTON VoSPI 输出的像素已是 uint16 BE，直接搬 |
| **波特率与帧率** | 921600 bps ÷ (38413×10) ≈ 2.4 fps 极限。2M 波特率 ≈ 6.5 fps |
| **平台相关** | 不同 MCU 的串口发送函数不同（ESP32 用 `Serial.write`，STM32 用 `HAL_UART_Transmit`），封装成 `uart_write()` 统一接口 |

## 四、WiFi vs 串口 对比

| 项目 | WiFi (TCP) | 串口 |
|------|-----------|------|
| **物理连接** | 无线，同一局域网即可 | USB 线 |
| **带宽** | 受 WiFi 信号限制，实测 2-20 Mbps | 921600 bps ≈ 0.9 Mbps |
| **帧率上限** | ~9 fps (LEPTON 原帧率) | ~2.4 fps |
| **延迟** | 1-10ms | < 1ms |
| **抗干扰** | TCP 自动重传保证完整 | 无校验层，依赖物理链路质量 |
| **上位机模式** | WiFi热成像 | 热成像(串口) |

## 五、上位机操作

### WiFi 模式

1. PC 和 ESP32 连接**同一 WiFi**
2. 打开上位机 → 工作模式选 **WiFi热成像**
3. 确认显示的本机 IP（如 `192.168.1.100`）
4. 端口默认 `8888`，点 **开始监听**
5. ESP32 上电 → 自动连接 WiFi → 连接 `192.168.1.100:8888`
6. 连接成功 → 状态栏显示 "ESP32 已连接" → 画面自动刷新

### 串口模式

1. 设备插 USB → 上位机工作模式选 **热成像(串口)**
2. 点刷新选 COM 口，CH340 会绿色高亮
3. 波特率选设备端配置的值（默认 921600）
4. 点 **打开串口** → 画面自动刷新

### 调试模式

1. 工作模式选 **串口调试** → 可用于查看 printf 日志
2. 接在**另一个串口**上（不要和热成像帧共用）
